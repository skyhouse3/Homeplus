CREATE OR REPLACE VIEW AD_GUESTLIST_VIEW
AS
    SELECT R.*, M.M_NAME
    FROM GUESTROOM R
    JOIN MEMBER M ON(R.M_CODE = M.M_CODE)
    WHERE GUER_RE_STATUS = 'N'
    ORDER BY GUER_RE_dATE DESC;

CREATE OR REPLACE VIEW AD_REALIST_VIEW
AS
    SELECT R.*, M.M_NAME
    FROM READINGROOM R
    JOIN MEMBER M ON(R.M_CODE = M.M_CODE)
    WHERE REA_STATUS = 'N'
    ORDER BY REA_RE_DATE DESC;

CREATE OR REPLACE VIEW APSEMIDETAIL AS
SELECT APM_CODE,APM_DCODE,BB_CODE,BB_DCATEGORY,BB_STATUS FROM APARTMENTMAN LEFT OUTER JOIN BULLETINBOARD 
ON APARTMENTMAN.APM_CODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,3,3) AND APARTMENTMAN.APM_DCODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,7,4)
AND BULLETINBOARD.BB_STATUS != 'Y'
ORDER BY APM_CODE, APM_DCODE;

create or replace view elecbill_view
as 
    (select ht_code, (전기세 + round(전기세 * 0.1) + round(전기세 * 0.037)) electTax
    from (select ht_code, round((case when ht_usage <= 100 then ((60.7 * ht_usage) + 410) 
                        when ht_usage <= 200 then ((125.9 * (ht_usage - 100)) + 6980) 
                        when ht_usage <= 300 then ((187.9 * (ht_usage - 200)) + 20260)
                        when ht_usage <= 400 then ((280.6 * (ht_usage - 300)) + 41300) 
                        when ht_usage <= 500 then ((   417.7 * (ht_usage - 400)) + 72810) 
                        when ht_usage > 500 then ((709.5 * (ht_usage - 500)) + 120220)  end)) 전기세
            from hometax
            where ht_name = '전기' AND HT_STATUS = 'Y'));
            

CREATE OR REPLACE VIEW FAC_LIST AS
SELECT BB_CODE, M_CODE, BB_CATEGORY, BB_TITLE, BB_CONTENT, BB_DCATEGORY, BB_ANSWER, BB_STATUS, BB_ORIGINAME, BB_RENAME,
    BB_CREATEDATE, BB_ANWDATE
FROM BULLETINBOARD JOIN MEMBER USING(M_CODE);

CREATE OR REPLACE VIEW FAC_PERSENT AS
SELECT BB_CODE, COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '1/1' THEN 'CATE1' END) CATE1,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,1) = '2' THEN 'CATE2' END) CATE2,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,1) = '3' THEN 'CATE5' END) CATE3,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '3/2' THEN 'CATE6' END) CATE4,
                    BB_STATUS
                    FROM BULLETINBOARD WHERE BB_STATUS != 'Y' group by BB_CODE, BB_STATUS;

CREATE OR REPLACE VIEW FAC_PERSENT2 AS
SELECT BB_CODE, COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '2/1' THEN 'CATE1' END) CATE1,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '2/2' THEN 'CATE2' END) CATE2,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '2/3' THEN 'CATE5' END) CATE3,
                    BB_STATUS
                    FROM BULLETINBOARD WHERE BB_STATUS != 'Y' group by BB_CODE, BB_STATUS;
                    
CREATE OR REPLACE VIEW FAC_PERSENT3 AS
SELECT BB_CODE, COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '3/4' THEN 'CATE1' END) CATE1,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '3/5' THEN 'CATE2' END) CATE2,
                    COUNT(CASE WHEN SUBSTR(BB_DCATEGORY,1,3) = '3/6' THEN 'CATE5' END) CATE3,
                    BB_STATUS
                    FROM BULLETINBOARD WHERE BB_STATUS != 'Y' group by BB_CODE, BB_STATUS;

create or replace view gasbill_view
as
    (select ht_code, round((ht_usage * 15.9347)) gasTax
    from hometax
    where ht_name = '가스' AND HT_STATUS = 'Y');
    
  commit;

CREATE OR REPLACE VIEW MMS_VIEW
AS
    SELECT MMS_CODE, MMS_TITLE, M1.M_ID WRITER, M2.M_ID RECEIVER, MMS_CONTENT, MMS_DATE, WRITER_STATUS, RECEIVER_STATUS, M1.M_CODE WRITER_CODE, M2.M_CODE RECEIVER_CODE
    FROM MEMBER_MESSAGE MMS
    JOIN MEMBER M1 ON(MMS.MMS_WRITER = M1.M_CODE)
    JOIN MEMBER M2 ON(MMS.MMS_RECEIVER = M2.M_CODE);
    
CREATE OR REPLACE VIEW MY_BOARD_VIEW
AS 
    SELECT B_CODE, B_TITLE, M_ID, B_NAME
    FROM BOARD
    WHERE B_STATUS = 'Y'
    ORDER BY ROWNUM DESC;
  
CREATE OR REPLACE VIEW MY_MEMO_VIEW
AS
    SELECT *
    FROM MEMBER_MEMO
    WHERE MME_STATUS = 'Y'
    ORDER BY ROWNUM DESC;
    
CREATE OR REPLACE VIEW MY_MSG_VIEW
AS

    ORDER BY ROWNUM DESC;
    
CREATE OR REPLACE VIEW MY_REPLY_VIEW
AS
    SELECT R.*, B.B_CONTENT, B.B_NAME, B.B_TITLE
    FROM BOARD_REPLY R
    JOIN BOARD B ON(R.B_CODE = B.B_CODE)
    WHERE R_STATUS = 'Y'

CREATE OR REPLACE VIEW PARKAREAMAN AS
SELECT PK_CODE,APM_CODE, BB_CODE, BB_DCATEGORY,BB_STATUS FROM PARKAREA LEFT OUTER JOIN BULLETINBOARD
ON PARKAREA.APM_CODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,5,3) AND PARKAREA.PK_CODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,9,2)
AND BULLETINBOARD.BB_STATUS != 'Y' AND SUBSTR(BB_DCATEGORY,1,3) = '3/4';
COMMIT;

CREATE OR REPLACE VIEW READINGMAN AS
SELECT R_CODE,BB_CODE,BB_DCATEGORY,BB_STATUS FROM READINGROOMMAN LEFT OUTER JOIN BULLETINBOARD 
ON READINGROOMMAN.R_CODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,5,2) AND READINGROOMMAN.R_CATE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,1,3)
AND BULLETINBOARD.BB_STATUS != 'Y' ORDER BY R_CODE;
COMMIT;

CREATE OR REPLACE VIEW ROADDEATIL AS
SELECT ROAD.ROAD_CODE,BULLETINBOARD.BB_CODE, BULLETINBOARD.BB_STATUS FROM ROAD LEFT JOIN BULLETINBOARD
ON ROAD.ROAD_CODE = SUBSTR(BULLETINBOARD.BB_DCATEGORY,5,2) AND BULLETINBOARD.BB_CATEGORY = '1' AND SUBSTR(BULLETINBOARD.BB_DCATEGORY,1,3) = '3/5'
AND BB_STATUS != 'Y'
ORDER BY ROAD_CODE;

as 
    (select ht_code, (case when ht_usage <= 30 then ((360 * ht_usage) + (400 * ht_usage) + (170 * ht_usage) + 1080)

                                                when ht_usage > 50 then ((790 * (ht_usage - 50)) + (1420 * (ht_usage - 50)) + (170 * ht_usage) + 53480)  end) waterTax
    from hometax
    where ht_name = '수도' AND HT_STATUS = 'Y');
